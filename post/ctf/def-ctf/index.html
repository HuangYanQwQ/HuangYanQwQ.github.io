这题利用了自修改代码保护，在静态分析时隐藏真实逻辑；

定位到核心校验函数 sub_140002900 分析出它要求输入一个 8 位密码

使用基于常数 0x61C88647 的循环位移哈希算法进行验证

通过还原验证成功后的解密逻辑，将数据段的密文每位先减去 7，再与 5 字节循环密钥序列 13 37 C0 DE 42 进行异或得到 Flag

# pip install z3-solver -i https://pypi.tuna.tsinghua.edu.cn/simple
from z3 import *

def solve_password():
    s = Solver()
    chars = [BitVec(f'c_{i}', 32) for i in range(8)]
    for c in chars:
        s.add(c >= 32, c <= 126)
        
    # v4 = 305419896
    v4 = BitVecVal(0x12345678, 32)
    # 0x61C88647
    MAGIC = 0x61C88647

    for i in range(8):
        rol_v4 = LShR(v4, 32 - 5) | (v4 << 5)
        # v4 = (char ^ ROL4(v4, 5)) - MAGIC
        v4 = (chars[i] ^ rol_v4) - MAGIC

    # -1873306505
    s.add(v4 == 0x90559477)

def decrypt_flag():
    target = [
        0x6E, 0x4C, 0xFB, 0xC4, 0x78, 0x53, 0x65, 0xBB, 0x88, 0x3D, 
        0x82, 0x0B, 0xB5, 0x88, 0x3D, 0x2E, 0x63, 0xFA, 0x88, 0x32, 0x6E
    ]
    
    # v11 (13 37 C0 DE) + v12 (42)
    key = [0x13, 0x37, 0xC0, 0xDE, 0x42]
    
    flag_content = ""
    for i in range(len(target)):
        # (Data[i] - 7) ^ Key[i % 5]
        char_code = (target[i] - 7) ^ key[i % 5]
        flag_content += chr(char_code & 0xFF)
    
    full_flag = f"def{{{flag_content}}}"
    print(f"Flag: {full_flag}")
    return full_flag

if __name__ == "__main__":
    decrypt_flag()
